#Strategy for filling data
davg <- merge(act, d2 , by = "interval", all=TRUE)
# head(davg)
# names(davg)

  dfill <-davg %>%
      select(date, steps, interval, mean_steps) %>%
      mutate(step_filled = case_when(steps>=0 ~ steps,
                                     is.na(steps) == TRUE ~ as.integer(mean_steps)))
